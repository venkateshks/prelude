** Sample msmtprc file which worked
defaults
port 587
tls on
auth on
logfile ~/.msmtp.log
tls_trust_file /etc/ssl/cert.pem


account gmail1
host smtp.gmail.com
from ****@gmail.com
user ***@gmail.com
tls_starttls on
tls on
auth on
port 587
passwordeval gpg --no-tty -q -d ~/.msmtp.gpg


account gmail2
host smtp.gmail.com
from ****@gmail.com
user ****@gmail.com
tls_starttls on
tls on
auth on
port 587
passwordeval gpg --no-tty -q -d ~/.msmtp_gmail2.gpg

account default : gmail2
